mapping = {
    '2': 'ABC', '3': 'DEF', '4': 'GHI',
    '5': 'JKL', '6': 'MNO', '7': 'PQRS',
    '8': 'TUV', '9': 'WXYZ'
}

cipher = "8444447777_7446666633_444777726622244433668"
result = []

for group in cipher.split(''):
    i = 0
    while i < len(group):
        n = group[i]
        count = 1
        while i + count < len(group) and group[i + count] == n:
            count += 1
        letters = mapping[n]
        letter = letters[(count - 1) % len(letters)]
        result.append(letter)
        i += count
    result.append(' ')

print(''.join(result))
